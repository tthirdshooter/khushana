const cards = document.querySelectorAll('.split'); cards.forEach(card => { card.addEventListener('mousemove', (e) => { const rect = card.getBoundingClientRect(); // Calculate mouse position relative to the center of the card const x = e.clientX - rect.left; const y = e.clientY - rect.top; const centerX = rect.width / 2; const centerY = rect.height / 2; // Degree of tilt (adjust '10' to make it more or less dramatic) const rotateX = (centerY - y) / 15; const rotateY = (x - centerX) / 15; card.style.transform = `rotateX(${rotateX}deg) rotateY(${rotateY}deg)`; }); // Reset position when mouse leaves card.addEventListener('mouseleave', () => { card.style.transform = `rotateX(0deg) rotateY(0deg)`; }); });
